
CREATE TABLE pelicula(
    id SMALLINT  PRIMARY KEY NOT NULL,
    name VARCHAR(255) NOT NULL, 
    description VARCHAR(255) NOT NULL,  
    estreno DATETIME NOT NULL,
    recaudacion DOUBLE  NOT NULL,
    status TINYINT  NOT NULL
);
